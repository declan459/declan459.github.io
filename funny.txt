# === CONFIG ===
$imageUrl = "https://images.theconversation.com/files/465400/original/file-20220525-12-5rdcq0.jpg?ixlib=rb-4.1.0&rect=6%2C15%2C2111%2C1393&q=45&auto=format&w=754&fit=clip"
$imagePath = "C:\wallpaper.jpg"

$audioUrl = "https://declan459.github.io/Download(1).mp3"
$audioPath = "C:\song.mp3"

# === DOWNLOAD FILES ===
Invoke-WebRequest -Uri $imageUrl -OutFile $imagePath
Invoke-WebRequest -Uri $audioUrl -OutFile $audioPath

# === SET WALLPAPER ===
Add-Type @"
using System.Runtime.InteropServices;
public class Wallpaper {
    [DllImport("user32.dll", SetLastError = true)]
    public static extern bool SystemParametersInfo(int uAction, int uParam, string lpvParam, int fuWinIni);
}
"@

[Wallpaper]::SystemParametersInfo(20, 0, $imagePath, 3)

# === PLAY AUDIO IN LOOP ===
Add-Type -AssemblyName presentationCore

$player = New-Object System.Windows.Media.MediaPlayer
$player.Open([Uri]$audioPath)

# Loop playback
$player.MediaEnded += {
    $player.Position = [TimeSpan]::Zero
    $player.Play()
}

$player.Volume = 0.5
$player.Play()

Write-Host "Wallpaper set and music playing..."

# Keep script alive so music continues
while ($true) {
    Start-Sleep -Seconds 5
}
